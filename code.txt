        private void FormWeb_KeyUp(object sender, KeyEventArgs e)
        {
            if (e.KeyCode == Keys.Escape)
                ToggleFullScreen(false);
            if (e.KeyCode == Keys.F11)
                ToggleFullScreen(true);
